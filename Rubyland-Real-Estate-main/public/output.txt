    <script>
        // Navbar scroll effect
        window.addEventListener('scroll', function() {
            const navbar = document.querySelector('.navbar');
            if (window.scrollY > 100) {
                navbar.classList.add('scrolled');
            } else {
                navbar.classList.remove('scrolled');
            }
        });

        // Testimonial slider
        const testimonialContainer = document.querySelector('.testimonials-container');
        const testimonialCards = document.querySelectorAll('.testimonial-card');
        const prevBtn = document.querySelector('.prev-btn');
        const nextBtn = document.querySelector('.next-btn');
        let currentIndex = 0;

        function updateTestimonialSlider() {
            const cardWidth = testimonialCards[0].offsetWidth + 30;
            testimonialContainer.style.transform = `translateX(-${currentIndex * cardWidth}px)`;
        }

        prevBtn.addEventListener('click', () => {
            currentIndex = Math.max(0, currentIndex - 1);
            updateTestimonialSlider();
        });

        nextBtn.addEventListener('click', () => {
            currentIndex = Math.min(testimonialCards.length - 1, currentIndex + 1);
            updateTestimonialSlider();
        });

        // Auto slide testimonials
        let testimonialInterval;
        function startTestimonialAutoSlide() {
            testimonialInterval = setInterval(() => {
                currentIndex = (currentIndex + 1) % testimonialCards.length;
                updateTestimonialSlider();
            }, 5000);
        }

        // Pause auto slide on hover
        document.querySelector('.testimonials').addEventListener('mouseenter', () => {
            clearInterval(testimonialInterval);
        });

        document.querySelector('.testimonials').addEventListener('mouseleave', startTestimonialAutoSlide);

        startTestimonialAutoSlide();

        // Listings slider
        const listingsContainer = document.querySelector('.listings-container');
        const listingCards = document.querySelectorAll('.listing-card');
        const listingPrev = document.querySelector('.listing-prev');
        const listingNext = document.querySelector('.listing-next');
        let listingIndex = 0;

        function updateListingSlider() {
            const cardWidth = listingCards[0].offsetWidth + 30;
            listingsContainer.style.transform = `translateX(-${listingIndex * cardWidth}px)`;
        }

        listingPrev.addEventListener('click', () => {
            listingIndex = Math.max(0, listingIndex - 1);
            updateListingSlider();
        });

        listingNext.addEventListener('click', () => {
            listingIndex = Math.min(listingCards.length - 1, listingIndex + 1);
            updateListingSlider();
        });

        // Initialize sliders
        window.addEventListener('resize', () => {
            updateTestimonialSlider();
            updateListingSlider();
        });

        // Smooth scrolling for anchor links
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                const target = document.querySelector(this.getAttribute('href'));
                if (target) {
                    target.scrollIntoView({
                        behavior: 'smooth',
                        block: 'start'
                    });
                }
            });
        });

        // Hero section animations
        document.addEventListener('DOMContentLoaded', function() {
            setTimeout(() => {
                document.querySelector('.hero h1').style.animationDelay = '0s';
                document.querySelector('.hero p').style.animationDelay = '0.3s';
                document.querySelector('.hero-buttons').style.animationDelay = '0.6s';
            }, 500);
        });
    </script>
